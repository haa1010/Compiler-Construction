PROGRAM  EXAMPLE5;  (* Example 5 *)
TYPE T = INTEGER;
CONST N = 10;
VAR  A : ARRAY(. 10 .) OF T;
     I : INTEGER;

PROCEDURE INPUT;
BEGIN
  FOR I := 1 TO N DO
     A(.I.) := READI;
END;


FUNCTION FINDMAX : INTEGER;
VAR MAX : INTEGER;
MAX := A(.0.);
BEGIN
    FOR I := 1 TO N DO
     BEGIN
        If MAX < A(.I.) 
        Then MAX := A(.I.);
     END
FINDMAX := MAX;
END;


BEGIN
CALL INPUT;
VAR MAX : INTEGER;

MAX := FINDMAX;

CALL WRITEI(MAX);
CALL WRITELN; 

 FOR I := 1 TO N DO
     BEGIN
        If A(.I.) = MAX
        Then 
          BEGIN  
            CALL WRITEI(I);
            CALL WRITELN; 
          END  
     END
END.  (* Example 5 *)
